<script>
    var so = '';
    var dot = '.';
    var doiso = '';
    var result = 0;
    var math = '';
    var flag =true;
    var kq=0;

    function view(data){
        $('#result').html(data).text(data);
    }

    function calc(){
        result = eval(parseFloat(so)+math+parseFloat(doiso));
        view(result);
        kq=result;
        so = result;
        try{
            if(math == '/'){
                if(doiso == 0){
                    var result = "NaN";
                    view(result);
                    so = '';
                    math = '';
                    doiso = '';
                    kq=0;
                }
            }   
        }
        catch(err){
            alert(err);
        }    
    }

    function reset(){
        so = '';
        math = '';
        doiso = '';
        kq=0;
       
    }
    
    function parseNumber($number, $string) {
        return $number+$string;
    }

    function setSo(number){
        if(math == ''){
            so = parseNumber(so, number);
            view(so);
        }
        else{
            doiso = parseNumber(doiso, number);
            view(doiso);
        }
    }

    function setMath(operation){
        math = operation;
        doiso = 0;
        flag = true;
    }

    $('.dot').click(function(){
        var temp = $(this).data('val');
        setSo(temp);
        console.log(temp);
        flag = true;
    });

    $('.btn').click(function(){
        if(flag == true)
        {
            var temp = $(this).data('val');
            setSo(temp);
            console.log(temp);
        }
       else{
            if(kq == ''){
                var temp = $(this).data('val');
                setSo(temp);
                console.log(temp);
            }
            else{
                reset();
                var temp = $(this).data('val');
                setSo(temp);
            }
        }
    });

    $('.btn-math').click(function(){
        var temp = $(this).data('val');
        setMath(temp);
        flag = true;
    });

    $('.btn-result').click(function(){
        calc();
        flag = false;
    });

    $('#reset').click(function(){
        reset();
        view("");
    });
